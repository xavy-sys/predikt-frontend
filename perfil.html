    function loadFounderProgramStatus() {
      const foundersCounter = document.getElementById("foundersCounter");
      const founderProgressFill = document.getElementById("founderProgressFill");
      const founderSlotsText = document.getElementById("founderSlotsText");
      const founderButton = document.getElementById("founderButton");
      const founderNote = document.getElementById("founderNote");

      if (!foundersCounter || !founderProgressFill || !founderSlotsText || !founderButton || !founderNote) {

      }

      /*
        Fuente futura:
        founders.v_founder_program_public_status
